<script>
let immagini = {
    "Casual": [
        "https://source.unsplash.com/200x200/?casual,outfit",
        "https://source.unsplash.com/200x200/?jeans,style"
    ],
    "Elegante": [
        "https://source.unsplash.com/200x200/?elegant,outfit",
        "https://source.unsplash.com/200x200/?dress,heels"
    ],
    "Streetwear": [
        "https://source.unsplash.com/200x200/?streetwear",
        "https://source.unsplash.com/200x200/?hoodie,style"
    ],
    "Anni 80": [
        "https://source.unsplash.com/200x200/?80s,fashion",
        "https://source.unsplash.com/200x200/?retro,outfit"
    ]
};

function generaOutfit() {

    let stile = document.getElementById("stile").value;
    let gallery = document.getElementById("gallery");

    gallery.innerHTML = "";

    immagini[stile].forEach(src => {

        let card = document.createElement("div");
        card.className = "card";

        let img = document.createElement("img");
        img.src = src;

        let heart = document.createElement("div");
        heart.className = "heart";
        heart.innerHTML = "🤍";

        heart.onclick = function() {
            heart.innerHTML = "💗";

            let clone = card.cloneNode(true);
            clone.querySelector(".heart").remove();

            document.getElementById("preferiti").appendChild(clone);
        };

        card.appendChild(img);
        card.appendChild(heart);
        gallery.appendChild(card);
    });
}
</script>
